```typescript
import { db } from "../server/db";
import {
  users, communities, userCommunities, rucks,
  challenges, challengeParticipants, communityPosts,
  friendships, ruckLikes, ruckComments, notifications,
} from "../shared/schema";
import bcrypt from "bcryptjs";

async function seed() {
  console.log("Seeding database...");

  // Clear existing data (order matters due to FK constraints)
  await db.delete(notifications);
  await db.delete(ruckComments);
  await db.delete(ruckLikes);
  await db.delete(communityPosts);
  await db.delete(challengeParticipants);
  await db.delete(challenges);
  await db.delete(rucks);
  await db.delete(userCommunities);
  await db.delete(communities);
  await db.delete(users);

  const passwordHash = await bcrypt.hash("password123", 10);

  // 1. Insert users (7-10)
  // Use realistic rucking community names: "Jake Martinez", "Sarah Thompson", etc.
  // All get the same password hash for testing convenience
  // Set onboardingComplete: true, provide weight (150-220), gender, location

  // 2. Insert communities (4-5)
  // Examples: "Mile High Ruckers" (Denver), "Coastal Ruck Club" (San Diego),
  //   "Capital City Rucking" (DC), "Bayou Ruck Crew" (Houston)
  // Set realistic memberCount values (15-85), category from schema enum, location, createdBy = a user id

  // 3. Insert userCommunities (join users to communities)
  // Each user joins 1-3 communities

  // 4. Insert rucks (25+)
  // Realistic distances: 3000-15000 meters (2-10 miles)
  // Durations: 1800-7200 seconds (30 min to 2 hours)
  // Weights: 20-60 lbs
  // Spread createdAt dates over the past 30 days
  // Include some notes like "Morning ruck through the park", "Heavy ruck training"
  // Include routeCoordinates as JSON string with 5-10 lat/lng points for some rucks
  // Do NOT include base64 routeImageUrl — leave null or omit

  // 5. Insert challenges (2-3)
  // Types: "distance", "weight", "rucks" per schema enum
  // Realistic goals: 100000 meters (62 miles), 50 rucks, etc.
  // endDate: 2-4 weeks from now

  // 6. Insert challengeParticipants

  // 7. Insert friendships (mix of accepted and pending)
  // status: "accepted" or "pending"

  // 8. Insert communityPosts
  // postType: "text" or "ruck"
  // Some with referenceId pointing to a ruck id

  // 9. Insert ruckLikes and ruckComments on some rucks

  // 10. Insert a few notifications

  console.log("Seed complete.");
  process.exit(0);
}

seed().catch((err) => {
  console.error("Seed failed:", err);
  process.exit(1);
});
```

IMPORTANT: Use db.insert().values([...]) for batch inserts. Use gen_random_uuid() or hardcoded UUIDs. For hardcoded UUIDs, use a consistent naming pattern so FK relationships are clear. Alternatively, use returning() to capture generated IDs.

Recommended approach: Use hardcoded UUIDs for all entities so FK relationships can be set statically:
```typescript
const userIds = {
  jake: "seed-user-001",
  sarah: "seed-user-002",
  // ...
};
```

This avoids async ID resolution and makes the script simple and deterministic.
