```
DATABASE_URL=postgres://postgres.[project-ref]:[password]@aws-0-[region].pooler.supabase.com:5432/postgres

SUPABASE_URL=https://[project-ref].supabase.co
SUPABASE_SERVICE_ROLE_KEY=eyJ...

# OAuth - Server validates these at startup. Server will crash if missing.
GOOGLE_CLIENT_ID=your-google-client-id.apps.googleusercontent.com
APPLE_SERVICE_ID=com.your.service.id
# Or use APPLE_BUNDLE_ID instead of APPLE_SERVICE_ID (at least one required)
# APPLE_BUNDLE_ID=com.your.bundle.id

# CORS - Comma-separated list of allowed origins (required in production)
ALLOWED_ORIGINS=https://your-app.up.railway.app

# Server
NODE_ENV=production
PORT=5000

# Client (Expo - these ARE safe for client bundle)
# D-21: EXPO_PUBLIC_DOMAIN must point to your Railway URL, NOT the old Replit URL
EXPO_PUBLIC_DOMAIN=your-app.up.railway.app
EXPO_PUBLIC_GOOGLE_CLIENT_ID=your-google-client-id.apps.googleusercontent.com
```
